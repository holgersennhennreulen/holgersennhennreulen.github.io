From ?rgamma: The Gamma distribution with parameters \(\text{shape} = a\) and \(\text{scale} = s\) [\(\ldots\)]. The mean and variance are \[\begin{equation} E\left(X\right) = a\cdot s \end{equation}\] and \[\begin{equation} Var\left(X\right) = a\cdot s^{2}. \end{equation}\]

true_scale <- .5
set.seed(123456789)
R <- 9
sites <- LETTERS[1:16]
n_pairs_per_site <- rep(1, length(sites)) # sample(1:5, length(sites), replace = T)
df_list <- vector("list", R)
par(mfrow = c(3, 3), mar = c(4, 4, 0, 0))
for (r in 1:R) {
  df <- data.frame(site = NULL, pair = NULL, mu_k = NULL, mu_j = NULL,
                   stringsAsFactors = FALSE)
  mu_k <- rnorm(length(sites), sd = .1)
  for (k in 1:length(sites)) {
    df_k <- data.frame(site = LETTERS[k],
                       x = c("Control", "Treatment"),
                       mu_k = mu_k[k],
                       mu_x = c(0, 0.5),
                       stringsAsFactors = FALSE)
    df <- rbind(df, df_k)
  }
  df$true_expected_value <- exp(df$mu_k + df$mu_x)
  df$scale <- true_scale
  df$shape <- df$true_expected_value / df$scale
  df$y <- rgamma(n = nrow(df), shape = df$shape, scale = df$scale)
  df_list[[r]] <- df
  boxplot(y ~ x, data = df, frame = F, las = 1, xlab = "")
  ddf <- ddply(df, c("x"), summarize,
               mean_y = mean(y))
  lines(ddf$mean_y, lwd = 2, col = rgb(0, .6, .4))
  rm(df)
}
